119 void Interface::Unify(Interface* that, Zone* zone, bool* ok) {
120  if (this->forward_) return this->Chase()->Unify(that, zone, ok);
121  if (that->forward_) return this->Unify(that->Chase(), zone, ok);
122  ASSERT(this->forward_ == NULL);
123  ASSERT(that->forward_ == NULL);
124 
125  *ok = true;
126  if (this == that) return;
127  if (this->IsValue()) return that->MakeValue(ok);
128  if (that->IsValue()) return this->MakeValue(ok);
129 
130 #ifdef DEBUG
131  if (FLAG_print_interface_details) {
132  PrintF("%*s# Unifying...\n", Nesting::current(), "");
133  PrintF("%*sthis = ", Nesting::current(), "");
134  this->Print(Nesting::current());
135  PrintF("%*sthat = ", Nesting::current(), "");
136  that->Print(Nesting::current());
137  }
138 #endif
139 
140  // Merge the smaller interface into the larger, for performance.
141  if (this->exports_ != NULL && (that->exports_ == NULL ||
142  this->exports_->occupancy() >= that->exports_->occupancy())) {
143  this->DoUnify(that, ok, zone);
144  } else {
145  that->DoUnify(this, ok, zone);
146  }
147 
148 #ifdef DEBUG

160 void Interface::DoUnify(Interface* that, bool* ok, Zone* zone) {
161  ASSERT(this->forward_ == NULL);
162  ASSERT(that->forward_ == NULL);
163  ASSERT(!this->IsValue());
164  ASSERT(!that->IsValue());
165  ASSERT(*ok);
166 
167 #ifdef DEBUG
168  Nesting nested;
169 #endif
170 
171  // Try to merge all members from that into this.
172  ZoneHashMap* map = that->exports_;
173  if (map != NULL) {
174  for (ZoneHashMap::Entry* p = map->Start(); p != NULL; p = map->Next(p)) {
175  this->DoAdd(p->key, p->hash, static_cast<Interface*>(p->value), zone, ok);
176  if (!*ok) return;
177  }
178  }
179 
180  // If the new interface is larger than that's, then there were members in
181  // 'this' which 'that' didn't have. If 'that' was frozen that is an error.
182  int this_size = this->exports_ == NULL ? 0 : this->exports_->occupancy();
183  int that_size = map == NULL ? 0 : map->occupancy();
184  if (that->IsFrozen() && this_size > that_size) {
185  *ok = false;
186  return;
187  }
188 
189  // Merge instance.
190  if (!that->instance_.is_null()) {
191  if (!this->instance_.is_null() && *this->instance_ != *that->instance_) {
192  *ok = false;
193  return;
194  }
195  this->instance_ = that->instance_;
196  }
197 
198  // Merge interfaces.
199  this->flags_ |= that->flags_;
200  that->forward_ = this;
201 }
