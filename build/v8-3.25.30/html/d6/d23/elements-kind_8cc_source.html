175 bool IsMoreGeneralElementsKindTransition(ElementsKind from_kind,
176  ElementsKind to_kind) {
177  switch (from_kind) {
178  case FAST_SMI_ELEMENTS:
179  return to_kind != FAST_SMI_ELEMENTS;
180  case FAST_HOLEY_SMI_ELEMENTS:
181  return to_kind != FAST_SMI_ELEMENTS &&
182  to_kind != FAST_HOLEY_SMI_ELEMENTS;
183  case FAST_DOUBLE_ELEMENTS:
184  return to_kind != FAST_SMI_ELEMENTS &&
185  to_kind != FAST_HOLEY_SMI_ELEMENTS &&
186  to_kind != FAST_DOUBLE_ELEMENTS;
187  case FAST_HOLEY_DOUBLE_ELEMENTS:
188  return to_kind == FAST_ELEMENTS ||
189  to_kind == FAST_HOLEY_ELEMENTS;
190  case FAST_ELEMENTS:
191  return to_kind == FAST_HOLEY_ELEMENTS;
192  case FAST_HOLEY_ELEMENTS:
193  return false;
194  default:
195  return false;
196  }
197 }
