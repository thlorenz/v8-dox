255  RegisterState GetRegisterState(int register_index) {
256  ASSERT(register_index >= 0);
257  if (register_index >= kNumCachedRegisters) {
258  return STACKED;
259  } else {
260  if ((register_index % 2) == 0) {
261  return CACHED_LSW;
262  } else {
263  return CACHED_MSW;
264  }
265  }
