48 struct IdentifierStart {
49  static inline bool Is(uc32 c) {
50  switch (c) {
51  case '$': case '_': case '\\': return true;
52  default: return unibrow::Letter::Is(c);
53  }

58 struct IdentifierPart {
59  static inline bool Is(uc32 c) {
60  return IdentifierStart::Is(c)
61  || unibrow::Number::Is(c)
62  || c == 0x200C // U+200C is Zero-Width Non-Joiner.
63  || c == 0x200D // U+200D is Zero-Width Joiner.
64  || unibrow::CombiningMark::Is(c)
65  || unibrow::ConnectorPunctuation::Is(c);
66  }

70 // WhiteSpace according to ECMA-262 5.1, 7.2.
71 struct WhiteSpace {
72  static inline bool Is(uc32 c) {
73  return c == 0x0009 || // <TAB>
74  c == 0x000B || // <VT>
75  c == 0x000C || // <FF>
76  c == 0xFEFF || // <BOM>
77  // \u0020 and \u00A0 are included in unibrow::WhiteSpace.
78  unibrow::WhiteSpace::Is(c);
79  }
